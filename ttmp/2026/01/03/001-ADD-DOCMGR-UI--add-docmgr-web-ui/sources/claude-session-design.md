## Store Structure (RTK Slices)

store:
  slices:
    - search:
        state:
          mode: "docs" | "reverselookup" | "files"
          query: string
          filters:
            ticket: string?
            topics: string[]
            docType: string?
            status: string?
            file: string?
            dir: string?
            orderBy: "path" | "last_updated" | "rank"
            includeArchived: bool
            includeScripts: bool
            includeControlDocs: bool
          activeChips: FilterChip[]
          cursor: string?
          results: SearchResult[]
          total: number
          diagnostics: Diagnostic[]
          loading: bool
          error: Error?
        actions:
          - setMode(mode)
          - setQuery(query)
          - setFilter(key, value)
          - removeFilter(key)
          - clearFilters()
          - appendResults(results, cursor)
          - resetResults()
          - setDiagnostics(diagnostics)

    - workspace:
        state:
          status: WorkspaceStatus?
          lastRefreshed: timestamp?
          refreshing: bool
          health: "ok" | "error" | "unknown"
        actions:
          - setStatus(status)
          - setHealth(health)
          - setRefreshing(bool)
    
    - preview:
        state:
          selectedResult: SearchResult?
          open: bool
          loading: bool
        actions:
          - selectResult(result)
          - closePreview()
    
    - ui:
        state:
          showDiagnostics: bool
          showFilters: bool
          showKeyboardHelp: bool
          theme: "light" | "dark"
        actions:
          - toggleDiagnostics()
          - toggleFilters()
          - toggleKeyboardHelp()

## RTK Query (API Slice)

api:
  baseQuery: "<http://127.0.0.1:8787/api/v1>"
  
  endpoints:
    - healthCheck:
        query: GET /healthz

    - getWorkspaceStatus:
        query: GET /workspace/status

        tags: ["workspace"]

        mutation: POST /index/refresh
        returns: { refreshed: bool, indexedAt: string, docsIndexed: number }

          includeDiagnostics: bool
        returns: SearchDocsResponse
        tags: ["search"]
